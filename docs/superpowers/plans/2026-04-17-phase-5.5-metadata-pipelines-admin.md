> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Ship an operator-grade admin page at `/automation/pipelines` for the `dataEnginePipelines` platform setting — table + edit dialog with live VAST resolution status, per-pipeline test-lookup, seed-defaults / seed-missing flows, inline enable/disable toggle.

**Architecture:** Pure consumer of existing backend for reads/writes (`GET /dataengine/pipelines/active`, `PUT /platform/settings`). Adds two small control-plane routes: `GET /dataengine/pipelines/defaults` (wraps `FileSeedLoader`) and `GET /metadata/lookup` (reuses the existing `proxyToVastdbQuery` helper). Web-UI follows the `NamingTemplatesPage` pattern: single file, co-located dialog, Vitest + React Testing Library tests mocking the api module.

**Tech Stack:** Fastify (`tsx`, `node:test` + `app.inject`), React 18 + Vite + Tailwind CSS v4, Vitest + @testing-library/react. TypeScript everywhere.

// services/control-plane/src/routes/dataengine-pipelines-defaults.ts
/**
 * GET /api/v1/dataengine/pipelines/defaults
 *
 * Returns the canonical default pipeline list by re-reading the same
 * seed JSON that pipeline-seed.ts uses on first-boot. Single source of
 * truth — admins can call this to populate an empty `dataEnginePipelines`
 * setting without duplicating the list in the web-ui.
 */

import type { FastifyInstance } from "fastify";

import { sendError } from "../http/errors.js";
import { withPrefix } from "../http/routes.js";
import { errorEnvelopeSchema } from "../http/schemas.js";
import { FileSeedLoader } from "../data-engine/pipeline-seed.js";
import {
  validatePipelineConfigList,
  type DataEnginePipelineConfig,
} from "../data-engine/pipeline-config.js";

/** Injectable loader — tests override to avoid reading real files. */
export interface PipelineDefaultsLoader {
  load(): unknown;
}

let loaderOverride: PipelineDefaultsLoader | null = null;

/** Test helper: inject a fake loader. Pass `null` to restore the default. */
export function __setPipelineDefaultsLoaderForTests(
  loader: PipelineDefaultsLoader | null,
): void {
  loaderOverride = loader;
}

function getLoader(): PipelineDefaultsLoader {
  return loaderOverride ?? new FileSeedLoader();
}

const defaultsResponseSchema = {
  type: "object",
  required: ["pipelines"],
  properties: {
    pipelines: {
      type: "array",
      items: {
        type: "object",
        required: ["fileKind", "functionName", "extensions", "targetSchema", "targetTable", "sidecarSchemaId"],
        properties: {
          fileKind: { type: "string", enum: ["image", "video", "raw_camera"] },
          functionName: { type: "string" },
          extensions: { type: "array", items: { type: "string" } },
          targetSchema: { type: "string" },
          targetTable: { type: "string" },
          sidecarSchemaId: { type: "string" },
          displayLabel: { type: "string" },
          enabled: { type: "boolean" },
        },
      },
    },
  },
} as const;

export async function registerDataEnginePipelineDefaultsRoute(
  app: FastifyInstance,
  prefixes: string[],
): Promise<void> {
  for (const prefix of prefixes) {
    const opPrefix = prefix === "/api/v1" ? "v1" : "legacy";

    app.get(
      withPrefix(prefix, "/dataengine/pipelines/defaults"),
      {
        schema: {
          tags: ["dataengine"],
          operationId: `${opPrefix}GetDataEnginePipelineDefaults`,
          summary: "Canonical default pipeline list (same seed JSON as pipeline-seed.ts)",
          response: {
            200: defaultsResponseSchema,
            500: errorEnvelopeSchema,
          },
        },
      },
      async (request, reply) => {
        let raw: unknown;
        try {
          raw = getLoader().load();
        } catch (err) {
          const msg = err instanceof Error ? err.message : String(err);
          return sendError(request, reply, 500, "SEED_UNAVAILABLE", `could not load seed file: ${msg}`);
        }

        let validated: DataEnginePipelineConfig[];
        try {
          validated = validatePipelineConfigList(raw);
        } catch (err) {
          const msg = err instanceof Error ? err.message : String(err);
          return sendError(request, reply, 500, "SEED_UNAVAILABLE", `seed file failed validation: ${msg}`);
        }

        return reply.send({ pipelines: validated });
      },
    );
  }
}
```

// services/control-plane/src/routes/metadata-lookup-proxy.ts
/**
 * GET /api/v1/metadata/lookup?path=&schema=&table=
 *
 * Admin proxy over vastdb-query's /api/v1/metadata/lookup. Reuses the
 * existing proxyToVastdbQuery helper so the base URL (VASTDB_QUERY_URL
 * env) and 503 fallback stay consistent with sibling metadata routes.
 *
 * Used by the Metadata Pipelines admin page's per-pipeline test-lookup
 * tool. No rate limiting — trusted internal admin surface.
 */

import type { FastifyInstance } from "fastify";

import { sendError } from "../http/errors.js";
import { withPrefix } from "../http/routes.js";
import { errorEnvelopeSchema } from "../http/schemas.js";
import { proxyToVastdbQuery } from "./exr-metadata.js";

export type MetadataLookupProxy = (path: string) => Promise<{
  ok: boolean;
  status: number;
  data: unknown;
}>;

let proxyOverride: MetadataLookupProxy | null = null;

/** Test helper: inject a stub. Pass `null` to restore the default. */
export function __setMetadataLookupProxyForTests(p: MetadataLookupProxy | null): void {
  proxyOverride = p;
}

function getProxy(): MetadataLookupProxy {
  return proxyOverride ?? proxyToVastdbQuery;
}

interface LookupQuery {
  path?: string;
  schema?: string;
  table?: string;
}

const lookupQuerySchema = {
  type: "object",
  required: ["path", "schema", "table"],
  properties: {
    path: { type: "string", minLength: 1 },
    schema: { type: "string", minLength: 1 },
    table: { type: "string", minLength: 1 },
  },
} as const;

export async function registerMetadataLookupProxyRoute(
  app: FastifyInstance,
  prefixes: string[],
): Promise<void> {
  for (const prefix of prefixes) {
    const opPrefix = prefix === "/api/v1" ? "v1" : "legacy";

    app.get<{ Querystring: LookupQuery }>(
      withPrefix(prefix, "/metadata/lookup"),
      {
        schema: {
          tags: ["metadata"],
          operationId: `${opPrefix}AdminMetadataLookup`,
          summary: "Admin-only schema-agnostic lookup against vastdb-query",
          querystring: lookupQuerySchema,
          response: {
            503: errorEnvelopeSchema,
          },
        },
      },
      async (request, reply) => {
        const { path, schema, table } = request.query;
        if (!path || !schema || !table) {
          return sendError(request, reply, 400, "VALIDATION_ERROR", "path, schema, table all required");
        }

        const q = new URLSearchParams({ path, schema, table }).toString();
        const result = await getProxy()(`/api/v1/metadata/lookup?${q}`);

        if (result.status === 503) {
          const detail = (result.data as { detail?: string })?.detail ?? "unknown";
          return sendError(request, reply, 503, "LOOKUP_UNREACHABLE", `vastdb-query unreachable: ${detail}`);
        }

        return reply.status(result.status).send(result.data);
      },
    );
  }
}
```

export interface DataEnginePipelineDefaultsResponse {
  pipelines: DataEnginePipelineConfig[];
}

export async function fetchMetadataPipelineDefaults(
  options: { signal?: AbortSignal } = {},
): Promise<DataEnginePipelineConfig[]> {
  const url = `${API_BASE_URL}/api/v1/dataengine/pipelines/defaults`;
  let response: Response;
  try {
    response = await fetch(url, { headers: withAuth(), signal: options.signal });
  } catch (err) {
    if (err instanceof DOMException && err.name === "AbortError") throw err;
    throw new ApiRequestError(0, `network error: ${String(err)}`);
  }
  if (!response.ok) {
    let msg = `${response.status}`;
    try {
      const body = await response.json();
      msg = (body as { error?: { message?: string } }).error?.message ?? msg;
    } catch { /* fallthrough — keep status as the message */ }
    throw new ApiRequestError(response.status, msg);
  }
  const data = (await response.json()) as DataEnginePipelineDefaultsResponse;
  return data.pipelines;
}

export async function saveMetadataPipelines(
  pipelines: DataEnginePipelineConfig[],
): Promise<void> {
  const url = `${API_BASE_URL}/api/v1/platform/settings`;
  const response = await fetch(url, {
    method: "PUT",
    headers: withAuth({ "content-type": "application/json" }),
    body: JSON.stringify({ dataEnginePipelines: pipelines }),
  });
  if (!response.ok) {
    let msg = `${response.status}`;
    try {
      const body = await response.json();
      msg = (body as { error?: { message?: string } }).error?.message ?? msg;
    } catch { /* keep status */ }
    throw new ApiRequestError(response.status, msg);
  }
}

export interface MetadataLookupResult {

  const q = new URLSearchParams({ path: args.path, schema: args.schema, table: args.table }).toString();
  const url = `${API_BASE_URL}/api/v1/metadata/lookup?${q}`;
  let response: Response;
  try {
    response = await fetch(url, { headers: withAuth(), signal: args.signal });
  } catch (err) {
    if (err instanceof DOMException && err.name === "AbortError") throw err;
    throw new ApiRequestError(0, `network error: ${String(err)}`);
  }

interface EditPipelineDialogProps {
  target: DiscoveredPipeline;
  allRows: DiscoveredPipeline[];
  onCancel: () => void;
  onSaved: () => Promise<void>;
}

function EditPipelineDialog({ target, allRows, onCancel, onSaved }: EditPipelineDialogProps) {
  const [functionName, setFunctionName] = useState(target.config.functionName);
  const [extensionsCsv, setExtensionsCsv] = useState(target.config.extensions.join(", "));
  const [targetSchema, setTargetSchema] = useState(target.config.targetSchema);
  const [targetTable, setTargetTable] = useState(target.config.targetTable);
  const [sidecarSchemaId, setSidecarSchemaId] = useState(target.config.sidecarSchemaId);
  const [displayLabel, setDisplayLabel] = useState(target.config.displayLabel ?? "");
  const [enabled, setEnabled] = useState(target.config.enabled !== false);
  const [submitting, setSubmitting] = useState(false);
  const [error, setError] = useState<string | null>(null);

  const buildConfig = useCallback((): DataEnginePipelineConfig => ({
    fileKind: target.config.fileKind,
    functionName: functionName.trim(),
    extensions: extensionsCsv.split(",").map((s) => s.trim()).filter((s) => s.length > 0),
    targetSchema: targetSchema.trim(),
    targetTable: targetTable.trim(),
    sidecarSchemaId: sidecarSchemaId.trim(),
    displayLabel: displayLabel.trim() || undefined,
    enabled,
  }), [target, functionName, extensionsCsv, targetSchema, targetTable, sidecarSchemaId, displayLabel, enabled]);

  const handleSave = useCallback(async () => {
    setSubmitting(true);
    setError(null);
    try {
      const formConfig = buildConfig();
      const merged: DataEnginePipelineConfig[] = allRows.map((r) =>
        r.config.fileKind === target.config.fileKind ? formConfig : r.config,
      );
      await saveMetadataPipelines(merged);
      await onSaved();
    } catch (e) {
      setError(e instanceof Error ? e.message : "Save failed");
    } finally {
      setSubmitting(false);
    }
  }, [allRows, target, buildConfig, onSaved]);

  const canSubmit =
    functionName.trim().length > 0 &&
    extensionsCsv.trim().length > 0 &&
    targetSchema.trim().length > 0 &&
    targetTable.trim().length > 0 &&
    sidecarSchemaId.trim().length > 0;

  return (
    <div className="fixed inset-0 z-50 flex items-center justify-center bg-black/60"
         role="dialog" aria-modal="true"
         onClick={onCancel}>
      <Card className="w-[900px] max-w-[95vw] max-h-[92vh] overflow-auto"
            onClick={(e: React.MouseEvent) => e.stopPropagation()}>
        <h3 className="text-lg font-semibold mb-3">
          Edit {KIND_LABEL[target.config.fileKind as FileKind] ?? target.config.fileKind} pipeline
        </h3>

        {error && (
          <div className="mb-3 p-2 rounded bg-red-500/10 border border-red-500/30 text-sm text-red-400">
            {error}
          </div>
        )}

        <div className="grid gap-3">
          <FormRow label="File kind">
            <input type="text" value={target.config.fileKind} disabled
                   aria-label="File kind"
                   className="w-full px-3 py-2 rounded border border-[var(--color-ah-border-muted)] bg-[var(--color-ah-bg)] disabled:opacity-60" />
          </FormRow>
          <FormRow label="Function name">
            <input type="text" value={functionName}
                   aria-label="Function name"
                   onChange={(e) => setFunctionName(e.target.value)}
                   className="w-full px-3 py-2 rounded border border-[var(--color-ah-border-muted)] bg-[var(--color-ah-bg)] font-[var(--font-ah-mono)] text-sm" />
          </FormRow>
          <FormRow label="Extensions (comma-separated, leading dot, lowercase)">
            <input type="text" value={extensionsCsv}
                   aria-label="Extensions"
                   onChange={(e) => setExtensionsCsv(e.target.value)}
                   className="w-full px-3 py-2 rounded border border-[var(--color-ah-border-muted)] bg-[var(--color-ah-bg)] font-[var(--font-ah-mono)] text-sm" />
          </FormRow>
          <FormRow label="Target schema">
            <input type="text" value={targetSchema}
                   aria-label="Target schema"
                   onChange={(e) => setTargetSchema(e.target.value)}
                   className="w-full px-3 py-2 rounded border border-[var(--color-ah-border-muted)] bg-[var(--color-ah-bg)] font-[var(--font-ah-mono)] text-sm" />
          </FormRow>
          <FormRow label="Target table">
            <input type="text" value={targetTable}
                   aria-label="Target table"
                   onChange={(e) => setTargetTable(e.target.value)}
                   className="w-full px-3 py-2 rounded border border-[var(--color-ah-border-muted)] bg-[var(--color-ah-bg)] font-[var(--font-ah-mono)] text-sm" />
          </FormRow>
          <FormRow label="Sidecar schema id">
            <input type="text" value={sidecarSchemaId}
                   aria-label="Sidecar schema id"
                   onChange={(e) => setSidecarSchemaId(e.target.value)}
                   className="w-full px-3 py-2 rounded border border-[var(--color-ah-border-muted)] bg-[var(--color-ah-bg)] font-[var(--font-ah-mono)] text-sm" />
          </FormRow>
          <FormRow label="Display label (optional)">
            <input type="text" value={displayLabel}
                   aria-label="Display label"
                   onChange={(e) => setDisplayLabel(e.target.value)}
                   className="w-full px-3 py-2 rounded border border-[var(--color-ah-border-muted)] bg-[var(--color-ah-bg)]" />
          </FormRow>
          <FormRow label="Enabled">
            <label className="flex items-center gap-2 text-sm">
              <input type="checkbox" checked={enabled}
                     onChange={(e) => setEnabled(e.target.checked)} />
              <span>Active for metadata extraction</span>
            </label>
          </FormRow>
        </div>

        <div className="flex justify-end gap-2 mt-4">
          <Button variant="ghost" onClick={onCancel}>Cancel</Button>
          <Button variant="primary" onClick={() => void handleSave()}
                  disabled={submitting || !canSubmit}>
            {submitting ? "Saving…" : "Save changes"}
          </Button>
        </div>
      </Card>
    </div>
  );
}

function FormRow({ label, children }: { label: string; children: React.ReactNode }) {
  return (
    <label className="block">
      <span className="text-xs font-medium text-[var(--color-ah-text-muted)]">{label}</span>
      <div className="mt-1">{children}</div>
    </label>
  );
}
