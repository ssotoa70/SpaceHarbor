```typescript
/**
 * callWithRetryAndTiming — standardized retry + timing wrapper.
 *
 * Wraps any async operation with:
 *   - exponential backoff with optional ±50% jitter
 *   - per-attempt timeout ceiling (via AbortController)
 *   - caller-supplied shouldRetry predicate
 *   - external AbortSignal for mid-backoff cancellation
 *   - [timing] structured log line per attempt
 *   - retry_attempts_total{op,outcome} Prometheus counter
 *
 * Matches the infra/ module style (functional exports, no class unless
 * representing a distinct error type).
 */

import type { FastifyBaseLogger } from "fastify";

import { incrementCounter } from "./metrics.js";

export type RetryOutcome =
  | "ok"
  | "retry"
  | "non_retryable"
  | "exhausted"
  | "timeout";

export interface RetryOptions {
  /** Operation name — goes into [timing] log prefix and the metric's op label. */
  op: string;
  /** Number of attempts (default: 6). */
  maxAttempts?: number;
  /** Base backoff schedule in ms (default: [5000, 15000, 30000, 60000, 120000]). */
  backoffMs?: readonly number[];
  /** ±50% jitter (default: true). */
  jitter?: boolean;
  /** Per-attempt timeout ceiling in ms (default: none). */
  perAttemptTimeoutMs?: number;
  /** Predicate — false means stop retrying. Default: retry on any Error. */
  shouldRetry?: (err: unknown, attempt: number) => boolean;
  /** AbortSignal for mid-backoff cancellation. */
  signal?: AbortSignal;
  /** Fastify logger or compatible (has .info, .warn, .error). */
  log: FastifyBaseLogger;
}

/** Thrown when all attempts are exhausted. Preserves the last cause. */
export class RetryExhaustedError extends Error {
  readonly attempts: number;
  readonly op: string;
  readonly cause: unknown;
  constructor(op: string, attempts: number, cause: unknown) {
    super(`retry exhausted for op=${op} after ${attempts} attempts`);
    this.name = "RetryExhaustedError";
    this.op = op;
    this.attempts = attempts;
    this.cause = cause;
  }
}

const DEFAULT_BACKOFF_MS = [5000, 15000, 30000, 60000, 120000] as const;
const DEFAULT_MAX_ATTEMPTS = 6;

/** Test-only helper: no-op today; placeholder for future in-module counter
 *  state if we add local aggregation. Kept so tests can call it defensively. */
export function __resetRetryCounterForTests(): void {
  // Prometheus counters via infra/metrics.ts are process-global; tests
  // use incremental assertions rather than absolute values.
}

function sleepWithAbort(ms: number, signal?: AbortSignal): Promise<void> {
  return new Promise((resolve, reject) => {
    if (signal?.aborted) {
      const err = new Error("aborted");
      err.name = "AbortError";
      reject(err);
      return;
    }
    const timer = setTimeout(() => {
      signal?.removeEventListener("abort", onAbort);
      resolve();
    }, ms);
    const onAbort = () => {
      clearTimeout(timer);
      const err = new Error("aborted");
      err.name = "AbortError";
      reject(err);
    };
    signal?.addEventListener("abort", onAbort, { once: true });
  });
}

async function runWithTimeout<T>(
  fn: () => Promise<T>,
  timeoutMs: number | undefined,
): Promise<T> {
  if (!timeoutMs) return fn();
  return new Promise<T>((resolve, reject) => {
    const timer = setTimeout(() => {
      const err = new Error(`attempt exceeded ${timeoutMs}ms`);
      err.name = "TimeoutError";
      reject(err);
    }, timeoutMs);
    fn().then(
      (value) => { clearTimeout(timer); resolve(value); },
      (err) => { clearTimeout(timer); reject(err); },
    );
  });
}

function jitterAround(baseMs: number, jitter: boolean): number {
  if (!jitter) return baseMs;
  // ±50% around base.
  return Math.round(baseMs * (0.5 + Math.random()));
}

function emitTimingLine(
  log: FastifyBaseLogger,
  fields: {
    op: string;
    attempt: number;
    latencyMs: number;
    cumulativeMs: number;
    outcome: RetryOutcome;
    reason?: string;
  },
): void {
  const parts = [
    "[timing]",
    `op=${fields.op}`,
    `attempt=${fields.attempt}`,
    `latency_ms=${fields.latencyMs}`,
    `cumulative_ms=${fields.cumulativeMs}`,
    `status=${fields.outcome}`,
  ];
  if (fields.reason) {
    // Normalize reason — strip newlines, cap length for log hygiene.
    const reason = fields.reason.replace(/\s+/g, " ").slice(0, 180);
    parts.push(`reason=${reason}`);
  }
  log.info(parts.join(" "));
}

/**
 * Execute `fn` with retry + timing + metric emission.
 *
 * Invariants:
 *   - Emits exactly one [timing] log line per attempt.
 *   - Increments retry_attempts_total{op, outcome} per attempt.
 *   - On all-attempts-fail, throws RetryExhaustedError wrapping the last cause.
 *   - On shouldRetry=false, throws the original error directly (no wrapper).
 */
export async function callWithRetryAndTiming<T>(
  fn: () => Promise<T>,
  opts: RetryOptions,
): Promise<T> {
  const maxAttempts = opts.maxAttempts ?? DEFAULT_MAX_ATTEMPTS;
  const backoff = opts.backoffMs ?? DEFAULT_BACKOFF_MS;
  const jitter = opts.jitter ?? true;
  const shouldRetry = opts.shouldRetry ?? (() => true);
  const startedAt = Date.now();
  let lastError: unknown;

  for (let attempt = 1; attempt <= maxAttempts; attempt++) {
    if (opts.signal?.aborted) {
      const err = new Error("aborted before attempt");
      err.name = "AbortError";
      throw err;
    }
    const attemptStartedAt = Date.now();
    try {
      const result = await runWithTimeout(fn, opts.perAttemptTimeoutMs);
      const now = Date.now();
      emitTimingLine(opts.log, {
        op: opts.op,
        attempt,
        latencyMs: now - attemptStartedAt,
        cumulativeMs: now - startedAt,
        outcome: "ok",
      });
      incrementCounter("retry_attempts_total", { op: opts.op, outcome: "ok" });
      return result;
    } catch (err) {
      lastError = err;
      const now = Date.now();
      const isTimeout = (err as Error).name === "TimeoutError";
      const willRetry = attempt < maxAttempts && shouldRetry(err, attempt);
      const outcome: RetryOutcome = !willRetry
        ? (attempt >= maxAttempts ? "exhausted" : "non_retryable")
        : (isTimeout ? "timeout" : "retry");
      emitTimingLine(opts.log, {
        op: opts.op,
        attempt,
        latencyMs: now - attemptStartedAt,
        cumulativeMs: now - startedAt,
        outcome,
        reason: (err as Error)?.message,
      });
      incrementCounter("retry_attempts_total", { op: opts.op, outcome });
      if (!willRetry) {
        if (attempt >= maxAttempts) {
          throw new RetryExhaustedError(opts.op, attempt, err);
        }
        throw err;
      }
      // Backoff (indexed by attempt-1, saturates at last element).
      const baseMs = backoff[Math.min(attempt - 1, backoff.length - 1)];
      await sleepWithAbort(jitterAround(baseMs, jitter), opts.signal);
    }
  }
  // Unreachable, but keeps TS happy.
  throw new RetryExhaustedError(opts.op, maxAttempts, lastError);
}
```

```typescript
/**
 * function-configs-store — typed read/write with 60s module cache.
 *
 * Module-level cache keyed by scope. Same shape as
 * hooks/useDataEnginePipelines.ts in web-ui — cache/inFlight/cacheTtlMs.
 *
 * Validates type + min/max on every write. Throws typed errors
 * (ValidationError, NotFoundError) that the route layer maps to 400/404.
 *
 * DB access is abstracted behind a small adapter interface so tests can
 * substitute an in-memory implementation without spinning up a VAST
 * connection. The production adapter wraps TrinoClient.
 */

import { TrinoClient } from "../db/trino-client.js";

export type ValueType = "int" | "float" | "bool" | "string" | "duration_seconds";

export interface FunctionConfigRow {
  scope: string;
  key: string;
  value_type: ValueType;
  value_json: string;
  default_json: string;
  min_json: string | null;
  max_json: string | null;
  description: string;
  label: string;
  category: string;
  last_edited_by: string | null;
  last_edited_at: string | null;
}

export interface FunctionConfig {
  scope: string;
  key: string;
  valueType: ValueType;
  value: unknown;
  default: unknown;
  min?: number;
  max?: number;
  description: string;
  label: string;
  category: string;

  lastEditedAt: string | null;

export class ValidationError extends Error {
  constructor(message: string) { super(message); this.name = "ValidationError"; }
}
export class NotFoundError extends Error {
  constructor(message: string) { super(message); this.name = "NotFoundError"; }

export interface DbAdapter {
  readScope(scope: string): Promise<FunctionConfigRow[]>;
  readOne(scope: string, key: string): Promise<FunctionConfigRow | null>;
  writeValue(scope: string, key: string, valueJson: string, actor: string, at: Date): Promise<void>;

// Production adapter (Trino). Lazily constructed on first use.
// --------------------------------------------------------------------------
let productionAdapter: DbAdapter | null = null;
function getProductionAdapter(): DbAdapter {
  if (productionAdapter) return productionAdapter;
  const endpoint = process.env.VAST_DB_ENDPOINT ?? process.env.VAST_TRINO_ENDPOINT ?? "";
  const client = new TrinoClient({
    endpoint,
    accessKey: process.env.VAST_ACCESS_KEY ?? "",
    secretKey: process.env.VAST_SECRET_KEY ?? "",
  });
  const schema = `vast."spaceharbor/production"`;
  productionAdapter = {
    async readScope(scope): Promise<FunctionConfigRow[]> {
      const rs = await client.query(
        `SELECT scope, key, value_type, value_json, default_json, min_json, max_json,
                description, label, category, last_edited_by, last_edited_at
           FROM ${schema}.function_configs WHERE scope = ?`,
        [scope],
      );
      return rs.rows.map((r: unknown[]) => ({
        scope: r[0] as string, key: r[1] as string, value_type: r[2] as ValueType,
        value_json: r[3] as string, default_json: r[4] as string,
        min_json: r[5] as string | null, max_json: r[6] as string | null,
        description: r[7] as string, label: r[8] as string, category: r[9] as string,
        last_edited_by: r[10] as string | null,
        last_edited_at: r[11] as string | null,
      }));
    },
    async readOne(scope, key): Promise<FunctionConfigRow | null> {
      const rs = await client.query(
        `SELECT scope, key, value_type, value_json, default_json, min_json, max_json,
                description, label, category, last_edited_by, last_edited_at
           FROM ${schema}.function_configs WHERE scope = ? AND key = ?`,
        [scope, key],
      );
      if (rs.rows.length === 0) return null;
      const r = rs.rows[0];
      return {
        scope: r[0] as string, key: r[1] as string, value_type: r[2] as ValueType,
        value_json: r[3] as string, default_json: r[4] as string,
        min_json: r[5] as string | null, max_json: r[6] as string | null,
        description: r[7] as string, label: r[8] as string, category: r[9] as string,
        last_edited_by: r[10] as string | null,
        last_edited_at: r[11] as string | null,
      };
    },
    async writeValue(scope, key, valueJson, actor, at): Promise<void> {
      await client.query(
        `UPDATE ${schema}.function_configs
            SET value_json = ?, last_edited_by = ?, last_edited_at = ?
          WHERE scope = ? AND key = ?`,
        [valueJson, actor, at.toISOString(), scope, key],

// --------------------------------------------------------------------------
// Module cache (60s TTL, keyed by scope).
// --------------------------------------------------------------------------
interface CacheEntry { data: FunctionConfig[]; expiresAt: number }
const cache: Map<string, CacheEntry> = new Map();
let cacheTtlMs = 60_000;

/** Test-only: clears cache + resets TTL. */
export function __resetFunctionConfigsCacheForTests(): void {
  cache.clear();
  cacheTtlMs = 60_000;
}

/** Test-only: override TTL. */
export function __setFunctionConfigsCacheTtlForTests(ttlMs: number): void {
  cacheTtlMs = ttlMs;

function decode(row: FunctionConfigRow): FunctionConfig {
  const parseMaybeNumeric = (s: string | null): number | undefined => {
    if (s == null) return undefined;
    const n = Number(JSON.parse(s));
    return Number.isFinite(n) ? n : undefined;
  };
  return {
    scope: row.scope,
    key: row.key,
    valueType: row.value_type,
    value: JSON.parse(row.value_json),
    default: JSON.parse(row.default_json),
    min: parseMaybeNumeric(row.min_json),
    max: parseMaybeNumeric(row.max_json),
    description: row.description,
    label: row.label,
    category: row.category,
    lastEditedBy: row.last_edited_by,
    lastEditedAt: row.last_edited_at,
  };
}

export async function getScope(scope: string): Promise<FunctionConfig[]> {
  const cached = cache.get(scope);
  if (cached && cached.expiresAt > Date.now()) return cached.data;
  const rows = await getAdapter().readScope(scope);
  const data = rows.map(decode);
  cache.set(scope, { data, expiresAt: Date.now() + cacheTtlMs });
  return data;
}

export async function getValue<T>(scope: string, key: string): Promise<T> {
  const all = await getScope(scope);
  const entry = all.find((c) => c.key === key);
  if (!entry) throw new NotFoundError(`config ${scope}.${key} not found`);
  return entry.value as T;
}

function validateOrThrow(row: FunctionConfigRow, value: unknown): unknown {
  switch (row.value_type) {
    case "bool": {
      if (typeof value !== "boolean") throw new ValidationError("value must be boolean");
      return value;
    }
    case "string": {
      if (typeof value !== "string") throw new ValidationError("value must be string");
      return value;
    }
    case "int": {
      const n = typeof value === "string" ? Number(value) : value;
      if (typeof n !== "number" || !Number.isInteger(n)) throw new ValidationError("value must be integer");
      const min = row.min_json ? Number(JSON.parse(row.min_json)) : undefined;
      const max = row.max_json ? Number(JSON.parse(row.max_json)) : undefined;
      if (min !== undefined && n < min) throw new ValidationError(`value must be >= ${min}`);
      if (max !== undefined && n > max) throw new ValidationError(`value must be <= ${max}`);
      return n;
    }
    case "float":
    case "duration_seconds": {
      const n = typeof value === "string" ? Number(value) : value;
      if (typeof n !== "number" || !Number.isFinite(n)) throw new ValidationError("value must be numeric");
      const min = row.min_json ? Number(JSON.parse(row.min_json)) : undefined;
      const max = row.max_json ? Number(JSON.parse(row.max_json)) : undefined;
      if (min !== undefined && n < min) throw new ValidationError(`value must be >= ${min}`);
      if (max !== undefined && n > max) throw new ValidationError(`value must be <= ${max}`);
      return n;
    }
  }
}

export async function setValue(scope: string, key: string, value: unknown, actor: string): Promise<void> {
  const adapter = getAdapter();
  const row = await adapter.readOne(scope, key);
  if (!row) throw new NotFoundError(`config ${scope}.${key} not found`);
  const validated = validateOrThrow(row, value);
  await adapter.writeValue(scope, key, JSON.stringify(validated), actor, new Date());
  cache.delete(scope);
}

export async function resetToDefault(scope: string, key: string, actor: string): Promise<void> {
  const row = await getAdapter().readOne(scope, key);
  if (!row) throw new NotFoundError(`config ${scope}.${key} not found`);
  const def = JSON.parse(row.default_json);
  await setValue(scope, key, def, actor);
}
```

```typescript
/**
 * function-configs admin routes.
 *
 *   GET  /api/v1/function-configs/:scope           — list configs in scope
 *   PUT  /api/v1/function-configs/:scope/:key      — set value (validated)
 *
 * Gated by IAM admin:system_config. All writes audited via sendAudit/audit-chain.
 */

import type { FastifyInstance } from "fastify";

import { sendError } from "../http/errors.js";
import { withPrefix } from "../http/routes.js";
import { errorEnvelopeSchema } from "../http/schemas.js";
import { requireIamPermission } from "../iam/require-permission.js";
import {
  getScope,
  setValue,
  NotFoundError,
  ValidationError,
  type FunctionConfig,
} from "../config/function-configs-store.js";

function toApi(c: FunctionConfig) {
  return {
    scope: c.scope,
    key: c.key,
    valueType: c.valueType,
    value: c.value,
    default: c.default,
    min: c.min,
    max: c.max,
    description: c.description,
    label: c.label,
    category: c.category,
    lastEditedBy: c.lastEditedBy,
    lastEditedAt: c.lastEditedAt,
  };
}

export async function registerFunctionConfigsRoutes(
  app: FastifyInstance,
  prefixes: string[],
): Promise<void> {
  for (const prefix of prefixes) {
    const opPrefix = prefix === "/api/v1" ? "v1" : "legacy";

    app.get<{ Params: { scope: string } }>(
      withPrefix(prefix, "/function-configs/:scope"),
      {
        preHandler: requireIamPermission("admin:system_config"),
        schema: {
          tags: ["admin"],
          operationId: `${opPrefix}GetFunctionConfigsScope`,
          summary: "List function configs in a scope",
          params: {
            type: "object",
            required: ["scope"],
            properties: { scope: { type: "string", minLength: 1 } },
          },
          response: { 503: errorEnvelopeSchema },
        },
      },
      async (request, reply) => {
        const { scope } = request.params;
        try {
          const configs = await getScope(scope);
          return reply.send({ configs: configs.map(toApi) });
        } catch (err) {
          return sendError(request, reply, 503, "DB_UNREACHABLE",
            `could not read function_configs: ${(err as Error).message}`);
        }
      },
    );

    app.put<{ Params: { scope: string; key: string }; Body: { value: unknown } }>(
      withPrefix(prefix, "/function-configs/:scope/:key"),
      {
        preHandler: requireIamPermission("admin:system_config"),
        schema: {
          tags: ["admin"],
          operationId: `${opPrefix}PutFunctionConfigsKey`,
          summary: "Set function config value (validated)",
          params: {
            type: "object",
            required: ["scope", "key"],
            properties: {
              scope: { type: "string", minLength: 1 },
              key: { type: "string", minLength: 1 },
            },
          },
          body: { type: "object", required: ["value"], properties: { value: {} } },
          response: { 503: errorEnvelopeSchema },
        },
      },
      async (request, reply) => {
        const { scope, key } = request.params;
        const { value } = request.body;
        const actor = request.user?.email ?? request.user?.sub ?? "unknown";
        try {
          await setValue(scope, key, value, actor);
        } catch (err) {
          if (err instanceof ValidationError) {
            return sendError(request, reply, 400, "VALIDATION_ERROR", err.message);
          }
          if (err instanceof NotFoundError) {
            return sendError(request, reply, 404, "CONFIG_KEY_NOT_FOUND", err.message);
          }
          return sendError(request, reply, 503, "DB_UNREACHABLE", (err as Error).message);
        }
        const configs = await getScope(scope);
        const updated = configs.find((c) => c.key === key);
        if (!updated) {
          return sendError(request, reply, 503, "DB_UNREACHABLE", "write succeeded but re-read missed");
        }
        return reply.send({ config: toApi(updated) });
      },
    );
  }
}
```

```typescript
import { useState, useMemo } from "react";
import type { FunctionConfig } from "../api";

export interface SettingCardProps {
  config: FunctionConfig;
  onSave: (newValue: unknown) => Promise<void>;
  onReset: () => Promise<void>;
}

export function SettingCard({ config, onSave, onReset }: SettingCardProps): JSX.Element {
  const [draft, setDraft] = useState<unknown>(config.value);
  const [error, setError] = useState<string | null>(null);
  const [saving, setSaving] = useState(false);

  const dirty = useMemo(() => {
    return JSON.stringify(draft) !== JSON.stringify(config.value);
  }, [draft, config.value]);

  const isAtDefault = useMemo(() => {
    return JSON.stringify(config.value) === JSON.stringify(config.default);
  }, [config.value, config.default]);

  const handleSave = async () => {
    setError(null);
    setSaving(true);
    try {
      await onSave(draft);
    } catch (err) {
      const e = err as { status?: number; message?: string };
      if (e.status === 400) {
        setError(e.message ?? "Invalid value");
      } else if (e.status === 403) {
        setError("You do not have permission to edit settings");
        setDraft(config.value);
      } else if (e.status === 503) {
        setError("Could not save — try again");
      } else {
        setError(e.message ?? "Save failed");
      }
    } finally {
      setSaving(false);
    }
  };

  const handleReset = async () => {
    setError(null);
    await onReset();
    setDraft(config.default);
  };

  const renderInput = () => {
    switch (config.valueType) {
      case "bool":
        return (
          <label className="inline-flex items-center gap-2 cursor-pointer">
            <input
              type="checkbox"
              checked={Boolean(draft)}
              onChange={(e) => setDraft(e.target.checked)}
              aria-label={config.label}
              className="h-4 w-4"
            />
            <span>{Boolean(draft) ? "Enabled" : "Disabled"}</span>
          </label>
        );
      case "duration_seconds":
        return (
          <div className="inline-flex items-center gap-1">
            <input
              type="number"
              value={Number(draft)}
              onChange={(e) => setDraft(Number(e.target.value))}
              min={config.min}
              max={config.max}
              aria-label={config.label}
              className="w-24 px-2 py-1 bg-[var(--color-ah-bg-raised)] border border-[var(--color-ah-border)] rounded"
            />
            <span className="text-xs text-[var(--color-ah-text-muted)]">s</span>
          </div>
        );
      case "int":
      case "float":
        return (
          <input
            type="number"
            value={Number(draft)}
            onChange={(e) => setDraft(Number(e.target.value))}
            min={config.min}
            max={config.max}
            step={config.valueType === "int" ? 1 : undefined}
            aria-label={config.label}
            className="w-24 px-2 py-1 bg-[var(--color-ah-bg-raised)] border border-[var(--color-ah-border)] rounded"
          />
        );
      case "string":
        return (
          <input
            type="text"
            value={String(draft)}
            onChange={(e) => setDraft(e.target.value)}
            aria-label={config.label}
            className="w-64 px-2 py-1 bg-[var(--color-ah-bg-raised)] border border-[var(--color-ah-border)] rounded"
          />
        );
    }
  };

  return (
    <div className="border border-[var(--color-ah-border)] rounded p-4 space-y-2">
      <div>
        <h4 className="text-sm font-semibold">{config.label}</h4>
        <p className="text-xs text-[var(--color-ah-text-muted)] mt-1">{config.description}</p>
      </div>
      <div>{renderInput()}</div>
      {error && (
        <p className="text-xs text-red-400" role="alert">{error}</p>
      )}
      <p className="text-xs text-[var(--color-ah-text-subtle)]">
        <span>default: {String(config.default)}</span>
        {config.min !== undefined && <span> · min: {config.min}</span>}
        {config.max !== undefined && <span> · max: {config.max}</span>}
        {!isAtDefault && config.lastEditedBy && (
          <span> · last edited by: {config.lastEditedBy}{config.lastEditedAt ? ` at ${new Date(config.lastEditedAt).toLocaleString()}` : ""}</span>
        )}
      </p>
      <div className="flex items-center gap-2">
        <button
          type="button"
          onClick={handleSave}
          disabled={!dirty || saving}
          className="px-3 py-1 text-xs bg-cyan-600 hover:bg-cyan-500 disabled:bg-gray-700 disabled:text-gray-500 rounded"
        >
          {saving ? "Saving…" : "Save"}
        </button>
        {!isAtDefault && (
          <button
            type="button"
            onClick={handleReset}
            className="px-3 py-1 text-xs bg-gray-700 hover:bg-gray-600 rounded"
          >
            Reset
          </button>
        )}
      </div>
    </div>
  );
}
```

```typescript
    const map = new Map<string, FunctionConfig[]>();

      map.set(c.category, bucket);

  const handleSave = async (key: string, value: unknown) => {
    await saveFunctionConfig(SCOPE, key, value);
    await refresh();

  const handleReset = async (config: FunctionConfig) => {
    await saveFunctionConfig(SCOPE, config.key, config.default);
    await refresh();
  };

  return (
    <section aria-label="Integrity settings" className="space-y-6">
      <header>
        <h1 className="text-xl font-bold">Integrity Settings</h1>
        <p className="text-sm text-[var(--color-ah-text-muted)] mt-1">
```
